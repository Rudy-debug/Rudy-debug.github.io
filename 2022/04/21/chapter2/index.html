Review of Vector Calculus

Posted on 2022-04-21 Edited on 2022-04-28 In fvm

Vectors and Vector Operations

\[ \begin{equation} \mathbf{v}=u \mathbf{i}+v \mathbf{j}+w \mathbf{k} \label{eq1} \end{equation} \]

\[ \begin{equation} \mathbf{v}=\left[\begin{array}{c}u \\ v \\ w\end{array}\right] \quad \mathbf{v}^{\mathrm{T}}=\left[\begin{array}{lll}u & v & w\end{array}\right] \label{eq2} \end{equation} \]

\[ \begin{equation} \|\mathbf{v}\|=\sqrt{\mathbf{v} \cdot \mathbf{v}}=\sqrt{u^{2}+v^{2}+w^{2}} \label{eq3} \end{equation} \]

\[ \begin{equation} \left.\begin{array}{l}\mathbf{v}_{1}=u_{1} \mathbf{i}+v_{1} \mathbf{j}+w_{1} \mathbf{k} \\ \mathbf{v}_{2}=u_{2} \mathbf{i}+v_{2} \mathbf{j}+w_{2} \mathbf{k}\end{array}\right\} \Rightarrow \mathbf{v}_{1}+\mathbf{v}_{2}=\left(u_{1}+u_{2}\right) \mathbf{i}+\left(v_{1}+v_{2}\right) \mathbf{j}+\left(w_{1}+w_{2}\right) \mathbf{k} \label{eq4} \end{equation} \]

\[ \begin{equation} \mathbf{v}_{1}=\left[\begin{array}{c}u_{1} \\ v_{1} \\ w_{1}\end{array}\right] \quad \mathbf{v}_{2}=\left[\begin{array}{l}u_{2} \\ v_{2} \\ w_{2}\end{array}\right] \Rightarrow \mathbf{v}_{1}+\mathbf{v}_{2}=\left[\begin{array}{c}u_{1}+u_{2} \\ v_{1}+v_{2} \\ w_{1}+w_{2}\end{array}\right] \label{eq5} \end{equation} \]

\[ \begin{equation} s \mathbf{v}=s(u \mathbf{i}+v \mathbf{j}+w \mathbf{k})=s u \mathbf{i}+s v \mathbf{j}+s w \mathbf{k}=\left[\begin{array}{c}s u \\ s v \\ s w\end{array}\right] \label{eq6} \end{equation} \]

The Dot Product of Two Vectors

\[ \begin{equation} \mathbf{v}_{1} \cdot \mathbf{v}_{2}=\left\|\mathbf{v}_{1}\right\|\left\|\mathbf{v}_{2}\right\| \cos \left(\mathbf{v}_{1}, \mathbf{v}_{2}\right) \label{eq7} \end{equation} \]

\[ \begin{equation} \begin{aligned} \mathbf{i} \cdot \mathbf{i}&=\mathbf{j} \cdot \mathbf{j}=\mathbf{k} \cdot \mathbf{k}=1 \\ \mathbf{i} \cdot \mathbf{j}=\mathbf{i} \cdot \mathbf{k}&=\mathbf{j} \cdot \mathbf{i}=\mathbf{j} \cdot \mathbf{k}=\mathbf{k} \cdot \mathbf{i}=\mathbf{k} \cdot \mathbf{j}=0 \end{aligned} \label{eq8} \end{equation} \]

\[ \begin{equation} \mathbf{v}_{1} \cdot \mathbf{v}_{2}=\left(u_{1} \mathbf{i}+v_{1} \mathbf{j}+w_{1} \mathbf{k}\right) \cdot\left(u_{2} \mathbf{i}+v_{2} \mathbf{j}+w_{2} \mathbf{k}\right)=u_{1} u_{2}+v_{1} v_{2}+w_{1} w_{2} \label{eq9} \end{equation} \]

The Unit Direction Vector

\[ \begin{equation} \left.\begin{array}{l}\mathbf{v} \cdot \mathbf{v}=\|\mathbf{v}\|\|\mathbf{v}\| \overbrace{\cos (\mathbf{v}, \mathbf{v})}^{=1}=\|\mathbf{v}\|^{2} \Rightarrow \mathbf{v} \cdot \frac{\mathbf{v}}{\|\mathbf{v}\|}=\|\mathbf{v}\| \\ \mathbf{v} \cdot \mathbf{e}_{\mathrm{v}}=\|\mathbf{v}\| \underbrace{\left\|\mathbf{e}_{\mathrm{v}}\right\|}_{=1} \underbrace{\cos \left(\mathbf{v}, \mathbf{e}_{\mathrm{v}}\right)}_{=1}=\|\mathbf{v}\| \Rightarrow \mathbf{v} \cdot \mathbf{e}_{\mathrm{v}}=\|\mathbf{v}\|\end{array}\right\} \Rightarrow \mathbf{e}_{\mathrm{v}}=\frac{\mathbf{v}}{\|\mathbf{v}\|} \label{eq10} \end{equation} \]

The Cross Product of Two Vectors

\[ \begin{equation} \left\|\mathbf{v}_{3}\right\|=\left\|\mathbf{v}_{1} \times \mathbf{v}_{2}\right\|=\left\|\mathbf{v}_{1}\right\|\left\|\mathbf{v}_{2}\right\| \sin \left(\mathbf{v}_{1}, \mathbf{v}_{2}\right) \label{eq11} \end{equation} \]

\[ \begin{equation} \begin{aligned} \mathbf{i} \times \mathbf{i}=\mathbf{j} \times \mathbf{j}=\mathbf{k} \times \mathbf{k}=0 \\ \mathbf{i} \times \mathbf{j}=\mathbf{k}=-\mathbf{j} \times \mathbf{i} \\ \mathbf{j} \times \mathbf{k}=\mathbf{i}=-\mathbf{k} \times \mathbf{j} \\ \mathbf{k} \times \mathbf{i}=\mathbf{j}=-\mathbf{i} \times \mathbf{k} \end{aligned} \label{eq12} \end{equation} \]

\[ \begin{equation} \begin{aligned} \mathbf{v}_{1} \times \mathbf{v}_{2}&=\left(u_{1} \mathbf{i}+v_{1} \mathbf{j}+w_{1} \mathbf{k}\right) \times\left(u_{2} \mathbf{i}+v_{2} \mathbf{j}+w_{2} \mathbf{k}\right) \\ &=u_{1} u_{2} \mathbf{i} \times \mathbf{i}+u_{1} v_{2} \mathbf{i} \times \mathbf{j}+u_{1} w_{2} \mathbf{i} \times \mathbf{k} \\ &+v_{1} u_{2} \mathbf{j} \times \mathbf{i}+v_{1} v_{2} \mathbf{j} \times \mathbf{j}+v_{1} w_{2} \mathbf{j} \times \mathbf{k} \\ &+w_{1} u_{2} \mathbf{k} \times \mathbf{i}+w_{1} v_{2} \mathbf{k} \times \mathbf{j}+w_{1} w_{2} \mathbf{k} \times \mathbf{k} \\ &=u_{1} u_{2} 0+u_{1} v_{2} \mathbf{k}+u_{1} w_{2}(-\mathbf{j}) \\ &+v_{1} u_{2}(-\mathbf{k})+v_{1} v_{2} 0+v_{1} w_{2} \mathbf{i} \\ &+w_{1} u_{2} \mathbf{j}+w_{1} v_{2}(-\mathbf{i})+w_{1} w_{2} 0 \\ &=\left(v_{1} w_{2}-v_{2} w_{1}\right) \mathbf{i}-\left(u_{1} w_{2}-u_{2} w_{1}\right) \mathbf{j}+\left(u_{1} v_{2}-u_{2} v_{1}\right) \mathbf{k} \end{aligned} \label{eq13} \end{equation} \]

\[ \begin{equation} \mathbf{v}_{1} \times \mathbf{v}_{2}=\left|\begin{array}{ccc}\mathbf{i} & \mathbf{j} & \mathbf{k} \\ u_{1} & v_{1} & w_{1} \\ u_{2} & v_{2} & w_{2}\end{array}\right|=\left[\begin{array}{c}v_{1} w_{2}-v_{2} w_{1} \\ u_{2} w_{1}-u_{1} w_{2} \\ u_{1} v_{2}-u_{2} v_{1}\end{array}\right] \label{eq14} \end{equation} \]

The Scalar Triple Product

\[ \begin{equation} \left(\mathbf{v}_{1} \cdot \left[\mathbf{v}_{2} \times \mathbf{v}_{3}\right]\right)= \left|\begin{array}{ccc}u_{1} & v_{1} & w_{1} \\ u_{2} & v_{2} & w_{2} \\ u_{3} & v_{3} & w_{3} \end{array}\right| \label{eq15} \end{equation} \]

Gradient of a Scalar and Directional Derivatives

\[ \begin{equation} \nabla=\frac{\partial}{\partial x} \mathbf{i}+\frac{\partial}{\partial y} \mathbf{j}+\frac{\partial}{\partial z} \mathbf{k} \label{eq16} \end{equation} \]

\[ \begin{equation} \nabla s=\frac{\partial s}{\partial x} \mathbf{i}+\frac{\partial s}{\partial y} \mathbf{j}+\frac{\partial s}{\partial z} \mathbf{k} \label{eq17} \end{equation} \]

\[ \begin{equation} \frac{d s}{d l}=\nabla s \cdot \mathbf{e}_{l}=\|\nabla s\| \cos \left(\nabla s, \mathbf{e}_{l}\right) \label{eq18} \end{equation} \]

Operations on the Nabla Operator

\[ \begin{equation} \nabla \cdot \mathbf{v}=\frac{\partial u}{\partial x}+\frac{\partial v}{\partial y}+\frac{\partial w}{\partial z} \label{eq19} \end{equation} \]

\[ \begin{equation} \nabla \cdot\left(\nabla s\right)=\nabla^{2} s=\frac{\partial^{2} s}{\partial x^{2}}+\frac{\partial^{2} s}{\partial y^{2}}+\frac{\partial^{2} s}{\partial z^{2}} \label{eq20} \end{equation} \]

A third type of vector product for multiplying two vectors, known as the dyadic product (eg. \(\mathbf{v}\mathbf{v}\)), and resulting in a tensor with its components formed by ordered pairs of the two vectors.

\[ \begin{equation} \nabla \cdot\left\{\nabla \mathbf{v}\right\}=\nabla^{2} \mathbf{v}=\left(\nabla^{2} u\right) \mathbf{i}+\left(\nabla^{2} v\right) \mathbf{j}+\left(\nabla^{2} w\right) \mathbf{k} \label{eq21} \end{equation} \]

\[ \begin{equation} \begin{aligned} \nabla \times \mathbf{v}&=\left(\frac{\partial}{\partial x} \mathbf{i}+\frac{\partial}{\partial y} \mathbf{j}+\frac{\partial}{\partial z} \mathbf{k}\right) \times(u \mathbf{i}+v \mathbf{j}+w \mathbf{k}) \\ &=\left|\begin{array}{ccc}\mathbf{i} & \mathbf{j} & \mathbf{k} \\ \frac{\partial}{\partial x} & \frac{\partial}{\partial y} & \frac{\partial}{\partial z} \\ u & v & w\end{array}\right|=\left(\frac{\partial w}{\partial y}-\frac{\partial v}{\partial z}\right) \mathbf{i}+\left(\frac{\partial u}{\partial z}-\frac{\partial w}{\partial x}\right) \mathbf{j}+\left(\frac{\partial v}{\partial x}-\frac{\partial u}{\partial y}\right) \mathbf{k} \end{aligned} \label{eq22} \end{equation} \]

\[ \begin{equation} \begin{aligned} \left[(\mathbf{v} \cdot \nabla) \mathbf{v}\right] &=(u \mathbf{i}+v \mathbf{j}+w \mathbf{k}) \cdot\left(\frac{\partial}{\partial x} \mathbf{i}+\frac{\partial}{\partial y} \mathbf{j}+\frac{\partial}{\partial z} \mathbf{k}\right)(u \mathbf{i}+v \mathbf{j}+w \mathbf{k}) \\ &=\left(u \frac{\partial}{\partial x}+v \frac{\partial}{\partial y}+w \frac{\partial}{\partial z}\right)(u \mathbf{i}+v \mathbf{j}+w \mathbf{k}) \\ &=\left(u \frac{\partial u}{\partial x}+v \frac{\partial u}{\partial y}+w \frac{\partial u}{\partial z}\right) \mathbf{i} +\left(u \frac{\partial v}{\partial x}+v \frac{\partial v}{\partial y}+w \frac{\partial v}{\partial z}\right) \mathbf{j} \\ &+\left(u \frac{\partial w}{\partial x}+v \frac{\partial w}{\partial y}+w \frac{\partial w}{\partial z}\right) \mathbf{k} \end{aligned} \label{eq23} \end{equation} \]

Additional Vector Operations

\[ \begin{equation} \nabla \cdot \left(\nabla \times \mathbf{v} \right)=0 \label{eq24} \end{equation} \]

\[ \begin{equation} \nabla \times \left(\nabla s \right)=0 \label{eq25} \end{equation} \]

\[ \begin{equation} \nabla \cdot \left(s \mathbf{v} \right)=s \nabla \cdot \mathbf{v}+\mathbf{v} \cdot \nabla s \label{eq26} \end{equation} \]

\[ \begin{equation} \nabla \times \left(s \mathbf{v} \right)=s \nabla \times \mathbf{v}+\nabla s \times \mathbf{v} \label{eq27} \end{equation} \]

\[ \begin{equation} \nabla\left(\mathbf{v}_{1} \cdot \mathbf{v}_{2}\right)=\mathbf{v}_{1} \times\left(\nabla \times \mathbf{v}_{2}\right)+\mathbf{v}_{2} \times\left(\nabla \times \mathbf{v}_{1}\right)+\left(\mathbf{v}_{1} \cdot \nabla\right) \mathbf{v}_{2}+\left(\mathbf{v}_{2} \cdot \nabla\right) \mathbf{v}_{1} \label{eq28} \end{equation} \]

\[ \begin{equation} \nabla \cdot\left(\mathbf{v}_{1} \times \mathbf{v}_{2}\right)=\mathbf{v}_{2} \cdot\left(\nabla \times \mathbf{v}_{1}\right)-\mathbf{v}_{1} \cdot\left(\nabla \times \mathbf{v}_{2}\right) \label{eq29} \end{equation} \]

\[ \begin{equation} \nabla \times\left(\mathbf{v}_{1} \times \mathbf{v}_{2}\right)=\mathbf{v}_{1}\left(\nabla \cdot \mathbf{v}_{2}\right)-\mathbf{v}_{2}\left(\nabla \cdot \mathbf{v}_{1}\right)+\left(\mathbf{v}_{2} \cdot \nabla\right) \mathbf{v}_{1}-\left(\mathbf{v}_{1} \cdot \nabla\right) \mathbf{v}_{2} \label{eq30} \end{equation} \]

\[ \begin{equation} \nabla \times(\nabla \times \mathbf{v})=\nabla(\nabla \cdot \mathbf{v})-\nabla^{2} \mathbf{v} \label{eq31} \end{equation} \]

\[ \begin{equation} (\nabla \times \mathbf{v}) \times \mathbf{v}=\mathbf{v} \cdot(\nabla \mathbf{v})-\nabla(\mathbf{v} \cdot \mathbf{v}) \label{eq32} \end{equation} \]

Matrices and Matrix Operations

A matrix \(\mathbf{A}\) of order \(M \times N\) is a rectangular array of quantities (\(M\) rows and \(N\) columns). An element of \(\mathbf{A}\) located on the \(i\)th row and \(j\)th column is denoted by \(a_{i j}\).

\[ \begin{equation} \mathbf{A}=\left[a_{i j}\right] \Rightarrow \mathbf{A}^{\mathrm{T}}=\left[a_{j i}\right] \label{eq33} \end{equation} \]

\[ \begin{equation} \mathbf{A}=\left[a_{i j}\right] \Rightarrow s \mathbf{A}=\left[s a_{j i}\right] \label{eq34} \end{equation} \]

To multiply two matrices \(\mathbf{A} (M \times N)\) and \(\mathbf{B} (N \times M)\), the number of columns of \(\mathbf{A}\) must be equal to the number of rows of \(\mathbf{B}\).

\[ \begin{equation} \begin{aligned} \mathbf{P}&=\mathbf{A}\mathbf{B} \\ p_{i j}&=\sum_{k=1}^{N} a_{i k} b_{k j} \end{aligned} \label{eq35} \end{equation} \]

Square Matrices

\[ \begin{equation} \mathbf{A}^{k}=\underbrace{\mathbf{A} \times \mathbf{A} \times \mathbf{A} \ldots \times \mathbf{A}}_{k \text { times }} \label{eq36} \end{equation} \]

\[ \begin{equation} \mathbf{A}^{-1} \mathbf{A}=\mathbf{A} \mathbf{A}^{-1}=\mathbf{I} \label{eq37} \end{equation} \]

\[ \begin{equation} \mathbf{U}=\left\{\begin{array}{cc}u_{i j} & i \leq j \\ 0 & i>j\end{array}\right. \label{eq38} \end{equation} \]

\[ \begin{equation} \mathbf{L}=\left\{\begin{array}{cc}\ell_{i j} & i \geq j \\ 0 & i<j\end{array}\right. \label{eq39} \end{equation} \]

Using Matrices to Describe Systems of Equations

\[ \begin{equation} \begin{aligned} a_{11} &\phi_{1}+a_{12} \phi_{2}+a_{13} \phi_{3}+\ldots+a_{1 N} \phi_{N}=b_{1} \\ a_{21} &\phi_{1}+a_{22} \phi_{2}+a_{23} \phi_{3}+\ldots+a_{2 N} \phi_{N}=b_{2} \\ a_{31} &\phi_{1}+a_{32} \phi_{2}+a_{33} \phi_{3}+\ldots+a_{3 N} \phi_{N}=b_{3} \\ &\vdots \quad \quad \quad \vdots \quad \quad \quad \vdots \quad \quad \quad \vdots \quad \quad \quad \vdots \quad \quad \quad \vdots \\ a_{N1} &\phi_{1}+a_{N2} \phi_{2}+a_{N3} \phi_{3}+\ldots+a_{N N} \phi_{N}=b_{N} \end{aligned} \label{eq40} \end{equation} \]

\[ \begin{equation} \left[\begin{array}{cccccc}a_{11} & a_{12} & a_{13} & \cdots & \cdots & a_{1 N} \\ a_{21} & a_{22} & a_{23} & \cdots & \cdots & a_{2 N} \\ a_{31} & a_{32} & a_{33} & \cdots & \cdots & a_{3 N} \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ a_{N 1} & a_{N 2} & a_{N 3} & \cdots & \cdots & a_{N N}\end{array}\right]\left[\begin{array}{c}\phi_{1} \\ \phi_{2} \\ \phi_{3} \\ \vdots \\ \vdots \\ \phi_{N}\end{array}\right]=\left[\begin{array}{c}b_{1} \\ b_{2} \\ b_{3} \\ \vdots \\ \vdots \\ b_{N}\end{array}\right] \label{eq41} \end{equation} \]

\[ \begin{equation} \mathbf{A} \boldsymbol{\phi}=\mathbf{b} \label{eq42} \end{equation} \]

The Determinant of a Square Matrix

\[ \begin{equation} \mathbf{A}=\left[\begin{array}{ll}a_{11} & a_{12} \\ a_{21} & a_{22}\end{array}\right] \Rightarrow \operatorname{det}(\mathbf{A})=a_{11} a_{22}-a_{21} a_{12} \label{eq43} \end{equation} \]

\[ \begin{equation} (c o)_{i j}=(-1)^{i+j}(m i)_{i j} \label{eq44} \end{equation} \]

A minor \((m i)_{i j}\) for an element \(a_{i j}\) is the determinant that results when the \(i\)th row and \(j\)th column are deleted. The cofactor \((c o)_{i j}\) of an element \(a_{i j}\) is the value of the minor multiplied by either a positive or a negative sign depending on whether \((i+j)\) is even or odd, respectively.

\[ \begin{equation} \operatorname{det}(\mathbf{A}) = \begin{cases}\sum_{i=1}^{N} a_{i j}(c o)_{i j} & \text { for any } j \\ \text { or } \\ \sum_{j=1}^{N} a_{i j}(c o)_{i j} & \text { for any } i\end{cases} \label{eq45} \end{equation} \]

Eigenvectors and Eigenvalues

In general the eigenvalues of a square matrix \(\mathbf{A}\) of order \(N\) are obtained from solving the following equation

\[ \begin{equation} \mathbf{A v}=\lambda \mathbf{v} \Rightarrow \mathbf{A} \mathbf{v}=\lambda \mathbf{I} \mathbf{v} \Rightarrow(\mathbf{A}-\lambda \mathbf{I}) \mathbf{v}=0 \label{eq46} \end{equation} \]

\[ \begin{equation} \mathbf{A}-\lambda \mathbf{I}=0 \Rightarrow \operatorname{det}(\mathbf{A}-\lambda \mathbf{I})=0 \label{eq47} \end{equation} \]

\[ \begin{equation} \operatorname{det}\left[\begin{array}{cccccc}a_{11}-\lambda & a_{12} & a_{13} & \cdots & \cdots & a_{1 N} \\ a_{21} & a_{22}-\lambda & a_{23} & \cdots & \cdots & a_{2 N} \\ a_{31} & a_{32} & a_{33}-\lambda & \cdots & \cdots & a_{3 N} \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ a_{N 1} & a_{N 2} & a_{N 3} & \cdots & \cdots & a_{N N}-\lambda\end{array}\right]=0 \label{eq48} \end{equation} \]

A Symmetric Positive-Definite Matrix

A symmetric matrix \(\mathbf{A}=\left[a_{ij}\right]\) of order \(N\) is positive-definite if for all column vectors \(\mathbf{p}\) in \(\mathbf{R}^N\) the following inequality holds:

\[ \begin{equation} \mathbf{p}^{\mathrm{T}}\mathbf{A}\mathbf{p}=0 \label{eq49} \end{equation} \]

\[ \begin{equation} \mathbf{A}=\left[\begin{array}{cccccc}a_{11} & a_{12} & a_{13} & \cdots & \cdots & a_{1 N} \\ a_{21} & a_{22} & a_{23} & \cdots & \cdots & a_{2 N} \\ a_{31} & a_{32} & a_{33} & \cdots & \cdots & a_{3 N} \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ \vdots & \vdots & \vdots & \vdots & \vdots & \vdots \\ a_{N 1} & a_{N 2} & a_{N 3} & \cdots & \cdots & a_{N N}\end{array}\right] \label{eq50} \end{equation} \]

\[ \begin{equation} \mathbf{P}=\left[\begin{array}{cccc}a_{11} & a_{12} & \cdots & a_{1 M} \\ a_{21} & a_{22} & \cdots & a_{2 M} \\ \vdots & \vdots & \vdots & \vdots \\ a_{M 1} & a_{M 2} & \cdots & a_{M M}\end{array}\right] \label{eq51} \end{equation} \]

Additional Matrix Operations

\[ \begin{equation} \begin{aligned} \mathbf{A}+(\mathbf{B}+\mathbf{C}) &=(\mathbf{A}+\mathbf{B})+\mathbf{C} \\ \mathbf{A}+\mathbf{B} &=\mathbf{B}+\mathbf{A} \\ s_{1}(\mathbf{A}+\mathbf{B}) &=s_{1} \mathbf{A}+s_{1} \mathbf{B} \\ \left(s_{1}+s_{2}\right) \mathbf{A} &=s_{1} \mathbf{A}+s_{2} \mathbf{A} \\ \mathbf{A}(\mathbf{B C}) &=(\mathbf{A} \mathbf{B}) \mathbf{C} \\ \mathbf{A I} &=\mathbf{I} \mathbf{A}=\mathbf{A} \\ \mathbf{A}(\mathbf{B}+\mathbf{C}) &=\mathbf{A} \mathbf{B}+\mathbf{A} \mathbf{C} \\ (\mathbf{A}+\mathbf{B}) \mathbf{C} &=\mathbf{A} \mathbf{C}+\mathbf{B C} \\ (\mathbf{A}+\mathbf{B})^{\mathrm{T}} &=\mathbf{A}^{\mathrm{T}}+\mathbf{B}^{\mathrm{T}} \\ \left(s_{1} \mathbf{A}\right)^{\mathrm{T}} &=s_{1} \mathbf{A}^{\mathrm{T}} \\ (\mathbf{A B})^{\mathrm{T}} &=\mathbf{B}^{\mathrm{T}} \mathbf{A}^{\mathrm{T}} \\ (\mathbf{A B})^{-1} &=\mathbf{B}^{-1} \mathbf{A}^{-1} \end{aligned} \label{eq52} \end{equation} \]

Tensors and Tensor Operations

\[ \begin{equation} \boldsymbol{\tau}=\left[\begin{array}{ccc}\tau_{x x} & \tau_{x y} & \tau_{x z} \\ \tau_{y x} & \tau_{y y} & \tau_{y z} \\ \tau_{z x} & \tau_{z y} & \tau_{z z}\end{array}\right] \quad \boldsymbol{\tau}^{\mathrm{T}}=\left[\begin{array}{ccc}\tau_{x x} & \tau_{y x} & \tau_{z x} \\ \tau_{x y} & \tau_{y y} & \tau_{z y} \\ \tau_{x z} & \tau_{y z} & \tau_{z z}\end{array}\right] \label{eq53} \end{equation} \]

\[ \begin{equation} \boldsymbol{\tau}=\mathbf{i i} \tau_{x x}+\mathbf{i} \mathbf{j} \tau_{x y}+\mathbf{i} \mathbf{k} \tau_{x z}+\mathbf{j i} \tau_{y x}+\mathbf{j} \mathbf{j} \tau_{y y}+\mathbf{j} \mathbf{k} \tau_{y z}+\mathbf{k i} \tau_{z x}+\mathbf{k j} \tau_{z y}+\mathbf{k} \mathbf{k} \tau_{z z} \label{eq54} \end{equation} \]

Note: \(\mathbf{i i}\) represents the dyadic product.

\[ \begin{equation} \left.\begin{array}{rl} \{\mathbf{v v}\}= & (u \mathbf{i}+v \mathbf{j}+w \mathbf{k})(u \mathbf{i}+v \mathbf{j}+w \mathbf{k}) \\ = & \mathbf{i i} u a+\mathbf{i j} u v+\mathbf{i} \mathbf{k} u w+ \\ & \mathbf{j i} v u+\mathbf{j} \mathbf{j} v v+\mathbf{j} \mathbf{k} v w+ \\ & \mathbf{k i} w u+\mathbf{k j} w v+\mathbf{k} \mathbf{k} w w \end{array}\right\} \Rightarrow\{\mathbf{v v}\}=\left[\begin{array}{ccc}u u & u v & u w \\ v u & v v & v w \\ w u & w v & w w\end{array}\right] \label{eq55} \end{equation} \]

\[ \begin{equation} \left.\begin{array}{rl} \{\nabla \mathbf{v}\}=&\left(\frac{\partial}{\partial x} \mathbf{i}+\frac{\partial}{\partial y} \mathbf{j}+\frac{\partial}{\partial z} \mathbf{k}\right)(u \mathbf{i}+v \mathbf{j}+w \mathbf{k}) \\ =&\mathbf{i i} \frac{\partial u}{\partial x}+\mathbf{i j} \frac{\partial v}{\partial x}+\mathbf{i k} \frac{\partial w}{\partial x}+ \\ &\mathbf{j i} \frac{\partial u}{\partial y}+\mathbf{j} \mathbf{j} \frac{\partial v}{\partial y}+\mathbf{j k} \frac{\partial w}{\partial y}+ \\ &\mathbf{k i} \frac{\partial u}{\partial z}+\mathbf{k j} \frac{\partial v}{\partial z}+\mathbf{k k} \frac{\partial w}{\partial z} \end{array}\right\} \Rightarrow\{\nabla \mathbf{v}\}=\left[\begin{array}{ccc} \frac{\partial u}{\partial x} & \frac{\partial v}{\partial x} & \frac{\partial w}{\partial x} \\ \frac{\partial u}{\partial y} & \frac{\partial v}{\partial y} & \frac{\partial w}{\partial y} \\ \frac{\partial u}{\partial z} & \frac{\partial v}{\partial z} & \frac{\partial w}{\partial z}\end{array}\right] \label{eq56} \end{equation} \]

Note: It is very important in the transfer process.

\[ \begin{equation} \boldsymbol{\Sigma}=\boldsymbol{\sigma}+\boldsymbol{\tau}=\left[\begin{array}{lll}\sigma_{x x}+\tau_{x x} & \sigma_{x y}+\tau_{x y} & \sigma_{x z}+\tau_{x z} \\ \sigma_{y x}+\tau_{y x} & \sigma_{y y}+\tau_{y y} & \sigma_{y z}+\tau_{y z} \\ \sigma_{z x}+\tau_{z x} & \sigma_{z y}+\tau_{z y} & \sigma_{z z}+\tau_{z z}\end{array}\right] \label{eq57} \end{equation} \]

\[ \begin{equation} \{s \boldsymbol{\tau}\}=\left[\begin{array}{ccc}s\tau_{x x} & s\tau_{x y} & s\tau_{x z} \\ s\tau_{y x} & s\tau_{y y} & s\tau_{y z} \\ s\tau_{z x} & s\tau_{z y} & s\tau_{z z}\end{array}\right] \label{eq58} \end{equation} \]

Fundamental Theorems of Vector Calculus

Gradient Theorem for Line Integrals

using \(\eqref{eq1}\) and \(\eqref{eq17}\), the vector \(\mathbf{r}\left(t\right)=\mathbf{r}\left[x\left(t\right),y\left(t\right),z\left(t\right)\right]\) for \(a < b\)

\[ \begin{equation} \begin{aligned} \int\limits_{C} \nabla s \cdot d \mathbf{r} =&\int\limits_{C}\left(\frac{\partial s}{\partial x} \mathbf{i}+\frac{\partial s}{\partial y} \mathbf{j}+\frac{\partial s}{\partial z} \mathbf{k}\right) \cdot \left(dx\left(t\right) \mathbf{i}+dy\left(t\right) \mathbf{j}+dz\left(t\right) \mathbf{k}\right) \\ =&\int_{a}^{b}\frac{\partial s}{\partial x}dx\left(t\right)+\int_{a}^{b}\frac{\partial s}{\partial y}dy\left(t\right)+\int_{a}^{b}\frac{\partial s}{\partial z}dz\left(t\right)\\ =&s(r(b))-s(r(a)) \end{aligned} \label{eq62} \end{equation} \]

Green's Theorem

Green's theorem expresses the contour integral of a simple closed curve \(C\) in terms of the double integral of the two dimensional region \(R\) bounded by \(C\).

\[ \begin{equation} \begin{aligned} \oint\limits_{C}(udx+vdy)=\iint\limits_{R}\left(\frac{\partial v}{\partial x}-\frac{\partial u}{\partial y}\right) dxdy \end{aligned} \label{eq63} \end{equation} \]

\[ \begin{equation} \begin{aligned} \left.\begin{array}{l} d\mathbf{r}=d x \mathbf{i}+d y \mathbf{j} \\ \mathbf{v}=u \mathbf{i}+v \mathbf{j} \\ d\mathbf{S}=d x d y \mathbf{k} \\ \end{array}\right\} \Rightarrow \\ \oint\limits_{C} \mathbf{v} \cdot d \mathbf{r}=\iint\limits_{R}\left[\nabla \times \mathbf{v}\right] \cdot d\mathbf{S} \end{aligned} \label{eq64} \end{equation} \]

Stokes' Theorem

Stokes' theorem is a higher dimensional version of Green's theorem.

\[ \begin{equation} \begin{aligned} \left.\begin{array}{l} d\mathbf{r}=d x \mathbf{i}+d y \mathbf{j} +dz \mathbf{k} \\ \mathbf{v}=u \mathbf{i}+v \mathbf{j} +w \mathbf{k} \\ d\mathbf{S}=dydz \mathbf{i} + dzdx \mathbf{j} +dxdy \mathbf{k}\\ \end{array}\right\} \\ \int\limits_{S}[\nabla \times \mathbf{v}] \cdot d \mathbf{S}=\oint\limits_{C} \mathbf{v} \cdot d \mathbf{r} \end{aligned} \label{eq65} \end{equation} \]

Divergence Theorem (Gauss' theorem)

\[ \begin{equation} \begin{aligned} dV=dxdydz \\ \int\limits_{V}\left(\nabla \cdot \mathbf{v}\right) dV&=\oint\limits_{S} \mathbf{v} \cdot \mathbf{n} dS \\ \int\limits_{V}\left[\nabla s\right] dV&=\oint\limits_{S} s d\mathbf{S} \\ \int\limits_{V}\left[\nabla \cdot \boldsymbol{\tau}\right] dV&=\oint\limits_{S} \left[\boldsymbol{\tau} \cdot \mathbf{n}\right] dS \end{aligned} \label{eq66} \end{equation} \]

Leibniz Integral Rule

\[ \begin{equation} \begin{aligned} \frac{d}{d t} \int_{a(t)}^{b(t)} \phi(x, t) dx=\underbrace{\int_{a(t)}^{b(t)} \frac{\partial \phi}{\partial t} dx}_{Term I}+\underbrace{\phi(b(t), t) \frac{\partial b}{\partial t}}_{\text {Term II}}-\underbrace{\phi(a(t), t) \frac{\partial a}{\partial t}}_{Term III} \end{aligned} \label{eq67} \end{equation} \]

\[ \begin{equation} \begin{aligned} \frac{d}{d t} \int\limits_{V(t)} \phi dV=\int\limits_{V(t)} \frac{\partial \phi}{\partial t} dV+\int\limits_{S(t)} \phi\left(\mathbf{v}_{s} \cdot \mathbf{n}\right) dS \end{aligned} \label{eq68} \end{equation} \]

\[ \begin{equation} \begin{aligned} \frac{d}{d t} \int\limits_{V} \phi d V=\int\limits_{V}\frac{\partial \phi}{\partial t} d V \end{aligned} \label{eq69} \end{equation} \]
